1   package org.hmis.cardGame;
2   
3   /*
4   An object of class card represents one of the 52 cards in a
5   standard deck of playing cards.  Each card has a suit and
6   a value.
7   Source: https://math.hws.edu/eck/cs124/javanotes4/c5/ex-5-5-answer.html
8    */
9   
10  
11  public class Card {
12  
13  	public final static int SPADES = 0,       // Codes for the 4 suits.
14  			HEARTS = 1,
15  			DIAMONDS = 2,
16  			CLUBS = 3;
17  
18  	public final static int ACE = 1,          // Codes for the non-numeric cards.
19  			JACK = 11,        //   Cards 2 through 10 have their 
20  			QUEEN = 12,       //   numerical values for their codes.
21  			KING = 13;
22  
23  	private final int suit;   // The suit of this card, one of the constants
24  	//    SPADES, HEARTS, DIAMONDS, CLUBS.
25  
26  	private final int value;  // The value of this card, from 1 to 13.
27  
28  	public Card(int theValue, int theSuit) {
29  		// Construct a card with the specified value and suit.
30  		// Value must be between 1 and 13.  Suit must be between
31  		// 0 and 3.  If the parameters are outside these ranges,
32  		// the constructed card object will be invalid.
33  		value = theValue;
34  		suit = theSuit;
35  	}
36  
37  	public int getSuit() {
38  		// Return the int that codes for this card's suit.
39  		return suit;
40  	}
41  
42  	public int getValue() {
43  		// Return the int that codes for this card's value.
44  		return value;
45  	}
46  
47  	public String getSuitAsString() {
48  		// Return a String representing the card's suit.
49  		// (If the card's suit is invalid, "??" is returned.)
50  		
51  		// https://unicode-table.com/en/sets/suits-of-the-cards/
52  		
53  		switch ( suit ) {
54  		case SPADES:   return "♠";
55  		case HEARTS:   return "♥";
56  		case DIAMONDS: return "♦";
57  		case CLUBS:    return "♣";
58  		default:       return "¯\\_(ツ)_/¯";
59  		}
60  	}
61  
62  	public String getValueAsString() {
63  		// Return a String representing the card's value.
64  		// If the card's value is invalid, "??" is returned.
65  		switch ( value ) {
66  		case 1:   return "A";
67  		case 2:   return "2";
68  		case 3:   return "3";
69  		case 4:   return "4";
70  		case 5:   return "5";
71  		case 6:   return "6";
72  		case 7:   return "7";
73  		case 8:   return "8";
74  		case 9:   return "9";
75  		case 10:  return "10";
76  		case 11:  return "J";
77  		case 12:  return "Q";
78  		case 13:  return "K";
79  		default:  return "¯\\_(ツ)_/¯";
80  		}
81  	}
82  
83  	public String toString() {
84  		// Return a String representation of this card, such as
85  		// "10 of Hearts" or "Queen of Spades".
86  		return getValueAsString() + "" + getSuitAsString();
87  	}
88  
89  
90  } // end class Card
